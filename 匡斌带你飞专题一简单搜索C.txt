//C - Catch That Cow
#include <iostream>
#include <cstdio>
#include <cstring>
#include <queue>
using namespace std;
const int Max = 100000 + 10;
int vis[Max];
int t[Max]; //¼ÇÂ¼Ê±¼ä
int n,k;

void bfs()
{
    queue<int>q;
    int temp;
    int i;
    int now;
    q.push(n);
    vis[n]=1;
    memset(t,0,sizeof(t));
    while(!q.empty())
    {
        now=q.front();
        q.pop();
        if(now==k)
        {
            cout<<t[now]<<endl;
            return;
        }
        else
        {
           for(i=0;i<3;i++)
           {
               if(i==0)
                temp=now+1;
               if(i==1)
                temp=now-1;
               if(i==2)
                temp=now*2;
               if(!vis[temp]&&temp>=0&&temp<Max)
               {
                   q.push(temp);
                   vis[temp]=1;
                   t[temp]=t[now]+1;
               }
           }

        }

    }
}

int main()
{
	while(scanf("%d %d",&n,&k) != EOF)
	{
		memset(vis, 0, sizeof(vis));
		bfs();
	}
	return 0;
}
